487 static void initializeViewMatrices()
488 {
489  Matrix cubeRotationMatrix = Matrix::createRotationY(MODEL_Y_ROTATION_ANGLE_IN_DEGREES);
490  Matrix cubeTranslationMatrix = Matrix::createTranslation(cubeGeometryProperties.position.x,
491  cubeGeometryProperties.position.y,
492  cubeGeometryProperties.position.z);
493  Matrix cubeModelMatrix = cubeRotationMatrix * cubeTranslationMatrix;
494  Matrix planeTranslationMatrix = Matrix::createTranslation(planeGeometryProperties.position.x,
495  planeGeometryProperties.position.y,
496  planeGeometryProperties.position.z);
497 
498  /* Initialize matrices that will be used to render geometry from camera point of view. */
499  {
500  cameraViewProperties.lookAtPoint.x = 0.0f;
501  cameraViewProperties.lookAtPoint.y = 0.0f;
502  cameraViewProperties.lookAtPoint.z = 0.0f;
503  cameraViewProperties.position.x = 0.0f;
504  cameraViewProperties.position.y = 0.0f;
505  cameraViewProperties.position.z = 20.0f;
506  cameraViewProperties.projectionMatrix = Matrix::matrixPerspective(degreesToRadians(CAMERA_PERSPECTIVE_FOV_IN_DEGREES),
507  float(windowWidth) / float(windowHeight),
508  NEAR_PLANE,
509  FAR_PLANE);
510  cameraViewProperties.upVector.x = 0.0f;
511  cameraViewProperties.upVector.y = 1.0f;
512  cameraViewProperties.upVector.z = 0.0f;
513  cameraViewProperties.viewMatrix = Matrix::matrixLookAt(cameraViewProperties.position,
514  cameraViewProperties.lookAtPoint,
515  cameraViewProperties.upVector);
516  cameraViewProperties.cubeViewProperties.modelMatrix = cubeModelMatrix;
517  cameraViewProperties.cubeViewProperties.modelViewMatrix = cameraViewProperties.viewMatrix *
518  cameraViewProperties.cubeViewProperties.modelMatrix;
519  cameraViewProperties.cubeViewProperties.modelViewProjectionMatrix = cameraViewProperties.projectionMatrix *
520  cameraViewProperties.cubeViewProperties.modelViewMatrix;
521 
522  cameraViewProperties.planeViewProperties.modelMatrix = planeTranslationMatrix;
523  cameraViewProperties.planeViewProperties.modelViewMatrix = cameraViewProperties.viewMatrix *
524  cameraViewProperties.planeViewProperties.modelMatrix;
525  cameraViewProperties.planeViewProperties.modelViewProjectionMatrix = cameraViewProperties.projectionMatrix *
526  cameraViewProperties.planeViewProperties.modelViewMatrix;
527 
528  Matrix cubeNormalMatrix = Matrix::matrixInvert(&cameraViewProperties.cubeViewProperties.modelViewMatrix);
529  Matrix planeNormalMatrix = Matrix::matrixInvert(&cameraViewProperties.planeViewProperties.modelViewMatrix);
530  Vec4f spotLightPosition = {spotLightProperties.position.x,
531  spotLightProperties.position.y,
532  spotLightProperties.position.z,
533  1.0f};
534 
535  Matrix::matrixTranspose(&cubeNormalMatrix);
536  Matrix::matrixTranspose(&planeNormalMatrix);
537 
538  cameraViewProperties.cubeViewProperties.normalMatrix = cubeNormalMatrix;
539  cameraViewProperties.planeViewProperties.normalMatrix = planeNormalMatrix;
540 
541  cameraViewProperties.spotLightPositionInEyeSpace = Matrix::vertexTransform(&spotLightPosition,
542  &cameraViewProperties.viewMatrix);
543  }
544 
545  /* Initialize matrices that will be used to render geometry from spot light point of view. */
546  {
547  lightViewProperties.cubeViewProperties.modelMatrix = cubeModelMatrix;
548  lightViewProperties.planeViewProperties.modelMatrix = planeTranslationMatrix;
549  lightViewProperties.lookAtPoint.x = 0.0f;

559  lightViewProperties.upVector.x = 0.0f;

562  }

566 static void renderFrame()
567 {
568  /* Spot light direction is changing during the rendering process: update it now. */
569  updateSpotLightDirection();
570 
571  /* Clear the contents of back buffer. */
572  GL_CHECK(glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT));
573 
574  /* Set uniform values that are common for all the rendering steps. */
575  GL_CHECK(glUniformMatrix4fv(renderSceneProgramLocations.uniformViewToColorTextureMatrix,
576  1,
577  GL_FALSE,
578  cameraViewProperties.viewToColorTextureMatrix.getAsArray()));
579  GL_CHECK(glUniformMatrix4fv(renderSceneProgramLocations.uniformViewToDepthTextureMatrix,
580  1,
581  GL_FALSE,
582  cameraViewProperties.viewToDepthTextureMatrix.getAsArray()));
583  GL_CHECK(glUniform4fv (renderSceneProgramLocations.uniformSpotLightLookAtPointInEyeSpace,
584  1,
585  (GLfloat*)&cameraViewProperties.spotLightLookAtPointInEyeSpace));
586  GL_CHECK(glUniform4fv (renderSceneProgramLocations.uniformSpotLightPositionInEyeSpace,
587  1,
588  (GLfloat*)&cameraViewProperties.spotLightPositionInEyeSpace));
589 
590  /* 1. Draw a scene from spot light point of view (calculate scene depth). */
591  {
592  /* Bind framebuffer object.
593  * There is a texture attached to depth attachment point for this framebuffer object.
594  * By using this framebuffer object, calculated depth values are stored in the texture. */
595  GL_CHECK(glBindFramebuffer(GL_FRAMEBUFFER, renderSceneObjects.framebufferObjectId));
596 
597  /* Clear the depth and colour buffers. */
598  GL_CHECK(glClear(GL_DEPTH_BUFFER_BIT | GL_COLOR_BUFFER_BIT));
599 
600  /* [Set viewport for light perspective] */
601  /* Set the view port to size of shadow map texture. */
602  GL_CHECK(glViewport(0, 0, shadowMapWidth, shadowMapHeight));
603  /* [Set viewport for light perspective] */
604 
605  /* [Enable shadow map drawing properties] */
606  GL_CHECK(glEnable(GL_POLYGON_OFFSET_FILL));
607  /* [Enable shadow map drawing properties] */
608 
609  /* Enable cull faceing to avoid self shadowing problems: only back faces of the geaometry will be rendered. */
610  GL_CHECK(glEnable(GL_CULL_FACE));
611 
612  /* [Set colour mask for shadow map rendering] */
613  /* Disable writing of each frame buffer colour component. */
614  GL_CHECK(glColorMask(GL_FALSE, GL_FALSE, GL_FALSE, GL_FALSE));
615  /* [Set colour mask for shadow map rendering] */
616 
617  /* [Draw the scene from spot light point of view] */
618  drawCubeAndPlane(false);
619  /* [Draw the scene from spot light point of view] */
620  } /* 1. */
621 
622  /* 2. Draw a scene with lights and shadows from eye point of view. */
623  {
624  /* Use the default framebuffer object: scene will be rendered on a screen. */
625  GL_CHECK(glBindFramebuffer(GL_FRAMEBUFFER, 0));
626 
627  /* Clear the depth and colour buffers. */
628  GL_CHECK(glClear(GL_DEPTH_BUFFER_BIT | GL_COLOR_BUFFER_BIT));
629 
630  /* Set the view port to the size of the window. */
631  GL_CHECK(glViewport(0, 0, windowWidth, windowHeight));
632 
633  GL_CHECK(glDisable(GL_CULL_FACE));
634  GL_CHECK(glDisable(GL_POLYGON_OFFSET_FILL));
635 
636  /* Enable writing of each frame buffer colour component. */
637  GL_CHECK(glColorMask(GL_TRUE, GL_TRUE, GL_TRUE, GL_TRUE));
638 
639  /* Draw the scene from camera point of view. */
640  drawCubeAndPlane(true);
641  } /* 2. */
642 }

647 {
648  /* Get triangular representation of the scene cube. Store the data in the cubeCoordinates array. */
649  CubeModel::getTriangleRepresentation(&cubeGeometryProperties.coordinates,
650  &cubeGeometryProperties.numberOfElementsInCoordinatesArray,

652 
653  /* Calculate normal vectors for the scene cube created above. */
654  CubeModel::getNormals(&cubeGeometryProperties.normals,
655  &cubeGeometryProperties.numberOfElementsInNormalsArray);
656 
657  /* Get triangular representation of a square to draw plane in XZ space. Store the data in the planeCoordinates array. */

659  &planeGeometryProperties.numberOfElementsInCoordinatesArray,
660  PLANE_SCALING_FACTOR);
661

665 
666  /* Fill buffer objects with data. */

934 /* Please see the specification above. */
935 static void updateSpotLightDirection()
936 {
937  /* Time used to set light direction and position. */
938  const float currentAngle = timer.getTime() / 4.0f;
939 
940  /* Update the look at point coordinates. */
941  lightViewProperties.lookAtPoint.x = SPOT_LIGHT_TRANSLATION_RADIUS * sinf(currentAngle);
942  lightViewProperties.lookAtPoint.y = -1.0f;
943  lightViewProperties.lookAtPoint.z = SPOT_LIGHT_TRANSLATION_RADIUS * cosf(currentAngle);
944 
945  /* Update all the view, projection matrixes athat are connected with updated look at point coordinates. */
946  Vec4f lookAtPoint = {lightViewProperties.lookAtPoint.x,
947  lightViewProperties.lookAtPoint.y,
948  lightViewProperties.lookAtPoint.z,
949  1.0f};
950 
951  /* Get lookAt matrix from the light's point of view, directed at the center of a plane.
952  * Store result in viewMatrixForShadowMapPass. */
953  lightViewProperties.viewMatrix = Matrix::matrixLookAt(lightViewProperties.position,
954  lightViewProperties.lookAtPoint,
955  lightViewProperties.upVector);
956 
957  lightViewProperties.cubeViewProperties.modelViewMatrix = lightViewProperties.viewMatrix * lightViewProperties.cubeViewProperties.modelMatrix;
958  lightViewProperties.planeViewProperties.modelViewMatrix = lightViewProperties.viewMatrix * lightViewProperties.planeViewProperties.modelMatrix;
959  lightViewProperties.cubeViewProperties.modelViewProjectionMatrix = lightViewProperties.projectionMatrix * lightViewProperties.cubeViewProperties.modelViewMatrix;
960  lightViewProperties.planeViewProperties.modelViewProjectionMatrix = lightViewProperties.projectionMatrix * lightViewProperties.planeViewProperties.modelViewMatrix;
961  cameraViewProperties.spotLightLookAtPointInEyeSpace = Matrix::vertexTransform(&lookAtPoint, &cameraViewProperties.viewMatrix);
962 
963  Matrix inverseCameraViewMatrix = Matrix::matrixInvert(&cameraViewProperties.viewMatrix);
964  /* [Define colour texture translation matrix] */
965  Matrix colorTextureTranslationMatrix = Matrix::createTranslation(COLOR_TEXTURE_TRANSLATION,
966  0.0f,
967  COLOR_TEXTURE_TRANSLATION);
968  /* [Define colour texture translation matrix] */
969 
970  /* [Calculate matrix for shadow map sampling: colour texture] */
971  cameraViewProperties.viewToColorTextureMatrix = Matrix::biasMatrix *
972  lightViewProperties.projectionMatrix *
973  lightViewProperties.viewMatrix *
974  colorTextureTranslationMatrix *
975  inverseCameraViewMatrix;
976  /* [Calculate matrix for shadow map sampling: colour texture] */
977  /* [Calculate matrix for shadow map sampling: depth texture] */
978  cameraViewProperties.viewToDepthTextureMatrix = Matrix::biasMatrix *
979  lightViewProperties.projectionMatrix *
980  lightViewProperties.viewMatrix *
981  inverseCameraViewMatrix;
982  /* [Calculate matrix for shadow map sampling: depth texture] */
983 }
