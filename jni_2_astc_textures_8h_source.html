132 /* Fragment shader source code. */
133 const char earth_fragment_shader_source[] =
134 {
135  "#version 300 es\n"
136  "precision mediump float;\n"
137  "uniform sampler2D cloud_texture;\n"
138  "uniform sampler2D daytime_texture;\n"
139  "uniform sampler2D nighttime_texture;\n"
140  "in vec2 tex2dcoord;\n"
141  "in vec3 normal;\n"
142  "in vec3 light;\n"
143  "in vec3 view;\n"
144  "out vec4 color;\n"
145  "void main() {\n"
146  " vec3 diffuse_albedo = vec3(2.0, 2.0, 3.0);\n"
147  " vec3 specular_albedo = vec3(0.1);\n"
148  " float specular_power = 16.0;\n"
149  " vec3 Normal = normalize(normal);\n"
150  " vec3 Light = normalize(light);\n"
151  " vec3 View = normalize(view);\n"
152  " vec3 reflected_light = reflect(-Light, Normal);\n"
153  " vec3 diffuse = max(dot(Normal, Light), 0.0) * diffuse_albedo;\n"
154  " vec3 specular = pow(max(dot(reflected_light, View), 0.0), specular_power) * specular_albedo;\n"
155  " vec2 clouds = texture(cloud_texture, tex2dcoord).rg;\n"
156  " vec3 daytime = (texture(daytime_texture, tex2dcoord).rgb * diffuse + specular * clouds.g) * (1.0 - clouds.r) + clouds.r * diffuse;\n"
157  " vec3 nighttime = texture(nighttime_texture, tex2dcoord).rgb * (1.0 - clouds.r) * 2.0;\n"
158  " color = vec4(mix(nighttime, daytime, 0.5), 1.0);\n"
159  "}\n"
160 };
